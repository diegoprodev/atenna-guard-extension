import { readFileSync, writeFileSync } from 'fs';
import { resolve } from 'path';

const MANIFEST_PATH = resolve('dist/manifest.json');
const LOCAL_PATTERNS = ['http://localhost/*', 'http://127.0.0.1/*'];

const manifest = JSON.parse(readFileSync(MANIFEST_PATH, 'utf-8'));

manifest.host_permissions = manifest.host_permissions.filter(
  (p) => !LOCAL_PATTERNS.includes(p)
);

if (manifest.content_scripts?.[0]?.matches) {
  manifest.content_scripts[0].matches = manifest.content_scripts[0].matches.filter(
    (p) => !LOCAL_PATTERNS.includes(p)
  );
}

writeFileSync(MANIFEST_PATH, JSON.stringify(manifest, null, 2));
console.log('[build] ✓ localhost removido do dist/manifest.json');
```

let _pending: Promise<unknown> | null = null;

export async function withRefreshLock<T>(fn: () => Promise<T>): Promise<T> {
  if (_pending) return _pending as Promise<T>;
  const timeout = new Promise<never>((_, reject) =>
    setTimeout(() => reject(new Error('refresh_timeout')), 10_000)
  );
  _pending = Promise.race([fn(), timeout]).finally(() => { _pending = null; });
  return _pending as Promise<T>;
}
```

# ── DLP event rate limiter (60 events/min per user, single-worker in-memory) ──
import collections, time as _time

_dlp_event_counts: dict[str, collections.deque] = {}

def _dlp_rate_limit_ok(user_id: str, limit: int = 60, window: int = 60) -> bool:
    now = _time.monotonic()
    dq = _dlp_event_counts.setdefault(user_id, collections.deque())

    if len(dq) >= limit:
