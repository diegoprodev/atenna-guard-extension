**Context:** `addGroupToHistory()` in `src/core/history.ts` stores the raw `question` string directly. If the user typed a CPF and clicked "send original", that CPF persists in chrome.storage.local indefinitely. The DLP patterns live in `src/dlp/patterns.ts`. Each pattern has shape:

The patterns use `/g` flag — reset `lastIndex` before use to avoid stateful RegExp issues.

```typescript
function maskPII(text: string): string {
  let result = text;
  for (const { type, pattern, validate } of DLP_PATTERNS) {
    const re = new RegExp(pattern.source, pattern.flags.includes('g') ? pattern.flags : pattern.flags + 'g');
    re.lastIndex = 0;
    result = result.replace(re, (match) => {
      if (validate && !validate(match)) return match;
      return `[${type}]`;
    });
  }
  return result;
}
```

```typescript
import { describe, it, expect, vi } from 'vitest';

// Minimal trapFocus extracted as a pure function for unit testing
function trapFocus(container: HTMLElement, onEscape: () => void): () => void {
  const focusable = () => Array.from(
    container.querySelectorAll<HTMLElement>(
      'button:not([disabled]), [href], input:not([disabled]), select:not([disabled]), textarea:not([disabled]), [tabindex]:not([tabindex="-1"])'
    )
  ).filter(el => el.offsetParent !== null);

  const onKeyDown = (e: KeyboardEvent) => {
    if (e.key === 'Escape') { onEscape(); return; }
    if (e.key !== 'Tab') return;
    const els = focusable();
    if (!els.length) return;
    const first = els[0];
    const last = els[els.length - 1];
    if (e.shiftKey) {
      if (document.activeElement === first) { e.preventDefault(); last.focus(); }
    } else {
      if (document.activeElement === last) { e.preventDefault(); first.focus(); }
    }
  };
  document.addEventListener('keydown', onKeyDown);
  return () => document.removeEventListener('keydown', onKeyDown);
}

describe('trapFocus', () => {
  function makeContainer(): HTMLElement {
    const div = document.createElement('div');
    div.innerHTML = '<button id="b1">B1</button><button id="b2">B2</button><button id="b3">B3</button>';
    document.body.appendChild(div);
    // jsdom doesn't implement offsetParent — override filter
    return div;
  }

  function makeKeyEvent(key: string, shiftKey = false): KeyboardEvent {
    return new KeyboardEvent('keydown', { key, shiftKey, bubbles: true, cancelable: true });
  }

  it('calls onEscape when Escape is pressed', () => {
    const container = makeContainer();
    const onEscape = vi.fn();
    const cleanup = trapFocus(container, onEscape);
    document.dispatchEvent(makeKeyEvent('Escape'));
    expect(onEscape).toHaveBeenCalledTimes(1);
    cleanup();
    document.body.removeChild(container);
  });

  it('cleanup removes the keydown listener', () => {
    const container = makeContainer();
    const onEscape = vi.fn();
    const cleanup = trapFocus(container, onEscape);
    cleanup();
    document.dispatchEvent(makeKeyEvent('Escape'));
    expect(onEscape).not.toHaveBeenCalled();
    document.body.removeChild(container);
  });

  it('is idempotent — calling cleanup twice does not throw', () => {
    const container = makeContainer();
    const cleanup = trapFocus(container, vi.fn());
    expect(() => { cleanup(); cleanup(); }).not.toThrow();
    document.body.removeChild(container);
  });
});
```

```typescript
async function isOnboarded(): Promise<boolean> {
  return new Promise(resolve => {
    chrome.storage.local.get('atenna_onboarded', r => resolve(!!r['atenna_onboarded']));
  });
}

async function markOnboarded(): Promise<void> {
  return new Promise(resolve => {
    chrome.storage.local.set({ atenna_onboarded: true }, resolve);
```

Modify `initPopup()` to gate on onboarding:

```typescript
export async function initPopup(): Promise<void> {
  const container = document.getElementById('atenna-popup')!;
  renderSkeleton(container);

  const [me, tabInfo, tabId] = await Promise.all([bffMe(), getActiveTabInfo(), getActiveTabId()]);

  if (!me) {
    renderLogin(container, tabId, tabInfo?.supported ?? false);
    return;
  }

  const onboarded = await isOnboarded();
  if (!onboarded) {
    renderFirstRunOnboarding(container, () => renderHome(container, me, tabInfo, tabId));
    return;
  }

  renderHome(container, me, tabInfo, tabId);
}
```
